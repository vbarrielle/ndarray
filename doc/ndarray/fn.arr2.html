pub fn arr2<A: Clone, V: ArrInit<A>>(xs: &[V]) -> Array<A, (Ix, Ix)>

Return a two-dimensional array with elements from xs.

Panics if the slices are not all of the same length.

use ndarray::arr2;

let a = arr2(&[[1, 2, 3],
               [4, 5, 6]]);
assert!(
    a.shape() == [2, 3]
);
